    <script>

        // 株価を取得する関数（Yahoo Financeのパブリックデータを利用）

        async function fetchStockPrice() {

            const symbol = 'VTI';

            try {

                // 今回はデモ用として、公式APIなしでも取得しやすいエンドポイントを想定

                const response = await fetch(`https://query1.finance.yahoo.com/v8/finance/chart/${symbol}`);

                const data = await response.json();

                const currentPrice = data.chart.result[0].meta.regularMarketPrice;

                

                document.getElementById('price').value = currentPrice;

                document.getElementById('last-update').innerText = `最終取得: ${new Date().toLocaleTimeString()} (15分遅延)` ;

                calculate();

            } catch (error) {

                console.error('価格取得失敗:', error);

                document.getElementById('last-update').innerText = '価格取得に失敗しました。手動入力してください。';

                document.getElementById('price').readOnly = false; // 失敗時は入力可能にする

            }

        }


        function calculate() {

            const price = document.getElementById('price').value;

            const quantity = document.getElementById('quantity').value;

            const total = price * quantity;

            // 小数点第2位まで表示

            document.getElementById('total').innerText = total.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2});

        }


        // ページを開いた時に株価を取得

        fetchStockPrice();

        // 5分ごとに自動更新

        setInterval(fetchStockPrice, 300000);

    </script>
